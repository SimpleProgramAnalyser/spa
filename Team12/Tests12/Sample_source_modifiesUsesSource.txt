procedure readPoint {
    if ((x < y) && (y < z)) then {
        cenX = cenX + readPoint;
        y = z + x;
    } else {
        y = x + y;
    }
    if ((x < y) && (y < z)) then {
        normSq = y;
        while (y < z) {
            normSq = z + y;
            print z;
        }
        print normSq;
    } else {
        y = x + y;
        read x;
        read y;
        call mainPrint;
    }
    while (x > z) {
        x = y;
        y = normSq + 3;
    }
}

procedure computeCentroid {
    count = 0;
    cenX = 0;
    cenY = 0;
    call readPoint;
    while ((x != 10) && (y != 100)) {
        count = count + 1;
        if (y < count) then {
            cenX = cenX + x;
            read count;
        } else {
            cenY = cenY + y;
        }
        while (y < x) {
            call readPoint;
        }
        cenY = cenX;
    }
    if (count == 0) then {
        flag = 1;
    } else {
        cenX = cenX / count;
        cenY = cenY / count;
        if (readPoint == 0) then {
            readPoint = 1;
        } else {
            readPoint = readPoint + 1;
        }
    }
    normSq = cenX * cenX + cenY * cenY;
}
