procedure main {    
     read steps;
     call raymarch;
     print depth; }

procedure raymarch {
    ro = 13;
    rd = 19;     
    read depth; 
    while (count < steps) {
        print depth;
        po = ro + rd * depth;
        call spheresdf;
        if (dist < epsilon) then {
            done = depth; }
        else {
            depth = depth + dist;} 
            count = count + 1; }}

procedure spheresdf {
    dist = x * x + y * y + z * z;
    x = dist;
    depth = depth;
    read p;
    while (x != p) {
        p = x;
        x = (dist / x + x) / 2; } 
    dist = x - 1;
    x = x * x + y * y / 2; }
